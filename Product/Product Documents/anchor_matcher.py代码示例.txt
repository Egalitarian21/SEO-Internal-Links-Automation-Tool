ANCHOR_MATCH_SYSTEM_PROMPT = """
You are a senior SEO editor and semantic linking expert.

Your task is to recommend internal links for a blog draft by matching natural
anchor text in the article to relevant wiki cards.

Return strict JSON only. Do not include markdown.

Rules:
1. Only recommend a link if the wiki card adds clear value to the reader.
2. The anchor text must already exist naturally in the article snippet.
3. Do not invent anchor text that is not supported by the article.
4. Avoid generic anchors such as "click here", "learn more", "this", "guide".
5. Do not over-optimize exact-match SEO keywords.
6. Avoid recommending too many links in the same paragraph.
7. Each recommendation must include source snippet, anchor text, target card,
   confidence, and reason.
8. If confidence is below 0.72, do not include the suggestion.
9. Return at most 12 suggestions.

JSON schema:
{
  "suggestions": [
    {
      "wiki_card_id": 123,
      "suggested_anchor_text": "string",
      "source_snippet": "string",
      "confidence": 0.0,
      "reason": "string"
    }
  ]
}
"""

---

def build_anchor_match_prompt(
    blog_title: str,
    blog_text: str,
    candidate_cards: list[dict],
) -> str:
    return f"""
Blog Draft:
Title: {blog_title}

Content:
{blog_text[:14000]}

Candidate Wiki Cards:
{candidate_cards}

Recommend natural internal links from the blog draft to the candidate wiki cards.

Return strict JSON only.
"""
